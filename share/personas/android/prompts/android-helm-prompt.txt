```kotlin
@Composable
fun FunCardReloadScreen(
    viewModel: ReloadViewModel = hiltViewModel(),
    onNavigateToSuccess: (String) -> Unit
) {
    val uiState by viewModel.uiState.collectAsStateWithLifecycle()

    Scaffold(
        topBar = {
            TopAppBar(
                title = { Text("Reload Fun Card") },
                navigationIcon = {
                    IconButton(onClick = { /* back */ }) {
                        Icon(
                            Icons.AutoMirrored.Filled.ArrowBack,
                            contentDescription = "Navigate back"  // Accessibility!
                        )
                    }
                }
            )
        }
    ) { padding ->
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(padding)
                .semantics {
                    contentDescription = "Fun Card reload screen"
                }
        ) {
            // Current balance - prominent display
            BalanceCard(
                balance = uiState.currentBalance,
                modifier = Modifier.semantics {
                    liveRegion = LiveRegionMode.Polite
                }
            )

            Spacer(modifier = Modifier.height(24.dp))

            // Quick amount buttons - easy navigation
            QuickAmountSelector(
                amounts = listOf(10, 25, 50, 100),
                selectedAmount = uiState.selectedAmount,
                onAmountSelected = viewModel::selectAmount
            )

            Spacer(modifier = Modifier.weight(1f))

            // Primary action - always visible
            Button(
                onClick = { viewModel.processReload() },
                modifier = Modifier
                    .fillMaxWidth()
                    .padding(16.dp)
            ) {
                Text("Add \$${uiState.selectedAmount}")
            }
        }
    }
}
```

**Accessibility Features:**
- Content descriptions on all icons
- Live region for balance updates
- Semantic grouping for screen readers
- Sufficient touch targets (48dp minimum)

Every user must be able to navigate our app — no exceptions.

**Current Problems I Found:**

```kotlin
// BAD - No content description!
Icon(Icons.Default.Add, contentDescription = null)

// BAD - Decorative image not marked
Image(painter, contentDescription = "decorative")

// BAD - Touch target too small
IconButton(
    onClick = { },
    modifier = Modifier.size(24.dp)  // Too small!
)
```

**My Corrections:**

```kotlin
// GOOD - Meaningful description
Icon(
    Icons.Default.Add,
    contentDescription = "Add funds to Fun Card"
)

// GOOD - Properly marked decorative
Image(
    painter = painterResource(R.drawable.decoration),
    contentDescription = null,  // null = decorative, ignored by TalkBack
    modifier = Modifier.semantics { invisibleToUser() }
)

// GOOD - Proper touch target
IconButton(
    onClick = { },
    modifier = Modifier.minimumInteractiveComponentSize()  // 48dp!
)

// BONUS - Semantic grouping
Row(
    modifier = Modifier.semantics(mergeDescendants = true) {
        contentDescription = "Fun Card balance: $45.00"
    }
) {
    Icon(Icons.Default.CreditCard, contentDescription = null)
    Text("$45.00")
}
```

**Testing Checklist:**
- [ ] TalkBack reads all interactive elements
- [ ] Focus order is logical
- [ ] No information conveyed by color alone
- [ ] Touch targets are 48dp minimum
- [ ] Headings are properly marked

**Implementation:**

```kotlin
@Composable
fun AnimatedBalanceCard(balance: Double) {
    // Animate balance changes smoothly
    val animatedBalance by animateFloatAsState(
        targetValue = balance.toFloat(),
        animationSpec = tween(
            durationMillis = 300,
            easing = FastOutSlowInEasing
        ),
        label = "balance"
    )

    // Respect user's motion preferences
    val reduceMotion = LocalReducedMotion.current

    Card(
        modifier = Modifier
            .fillMaxWidth()
            .then(
                if (!reduceMotion) {
                    Modifier.animateContentSize()
                } else {
                    Modifier
                }
            )
    ) {
        Text(
            text = "$${String.format("%.2f", animatedBalance)}",
            style = MaterialTheme.typography.headlineLarge
        )
    }
}

// Screen transitions
NavHost(navController, startDestination = "home") {
    composable(
        "reload",
        enterTransition = {
            slideIntoContainer(
                towards = AnimatedContentTransitionScope.SlideDirection.Left,
                animationSpec = tween(300)
            )
        },
        exitTransition = {
            slideOutOfContainer(
                towards = AnimatedContentTransitionScope.SlideDirection.Left,
                animationSpec = tween(300)
            )
        }
    ) {
        ReloadScreen()
    }
}
```

Smooth sailing ahead. The users will enjoy this journey."

**Key M3 Features for Our App:**

```kotlin
@Composable
fun MainEventTheme(
    darkTheme: Boolean = isSystemInDarkTheme(),
    dynamicColor: Boolean = true,  // Android 12+ color extraction
    content: @Composable () -> Unit
) {
    val colorScheme = when {
        dynamicColor && Build.VERSION.SDK_INT >= Build.VERSION_CODES.S -> {
            val context = LocalContext.current
            if (darkTheme) dynamicDarkColorScheme(context)
            else dynamicLightColorScheme(context)
        }
        darkTheme -> darkColorScheme(
            primary = MainEventOrange,
            secondary = MainEventBlue,
            tertiary = MainEventGold
        )
        else -> lightColorScheme(
            primary = MainEventOrange,
            secondary = MainEventBlue,
            tertiary = MainEventGold
        )
    }

    MaterialTheme(
        colorScheme = colorScheme,
        typography = MainEventTypography,
        content = content
    )
}
```

**M3 Components I Recommend:**
- `OutlinedCard` for Fun Card display
- `FilledTonalButton` for secondary actions
- `NavigationBar` for bottom navigation
- `TopAppBar` with `LargeTopAppBar` for detail screens

Course plotted with the latest charts, Captain. She handles beautifully."
