**ViewModel Test Suite:**

```kotlin
@OptIn(ExperimentalCoroutinesApi::class)
class FunCardReloadViewModelTest {

    @get:Rule
    val mainDispatcherRule = MainDispatcherRule()

    private lateinit var viewModel: FunCardReloadViewModel
    private val mockRepository: FunCardRepository = mockk()
    private val mockAnalytics: AnalyticsTracker = mockk(relaxed = true)

    @BeforeEach
    fun setup() {
        coEvery { mockRepository.getCurrentBalance() } returns Result.success(45.00)
        viewModel = FunCardReloadViewModel(mockRepository, mockAnalytics)
    }

    @Test
    fun `initial state loads current balance`() = runTest {
        viewModel.uiState.test {
            val state = awaitItem()
            assertThat(state.currentBalance).isEqualTo(45.00)
            assertThat(state.isLoading).isFalse()
        }
    }

    @Test
    fun `selectAmount updates selected amount`() = runTest {
        viewModel.selectAmount(50)

        viewModel.uiState.test {
            val state = awaitItem()
            assertThat(state.selectedAmount).isEqualTo(50)
        }
    }

    @Test
    fun `processReload success updates balance`() = runTest {
        coEvery { mockRepository.reloadCard(any()) } returns Result.success(95.00)

        viewModel.selectAmount(50)
        viewModel.processReload()

        viewModel.uiState.test {
            val state = awaitItem()
            assertThat(state.currentBalance).isEqualTo(95.00)
            assertThat(state.reloadSuccess).isTrue()
        }
    }

    @Test
    fun `processReload failure shows error`() = runTest {
        coEvery { mockRepository.reloadCard(any()) } returns
            Result.failure(IOException("Network error"))

        viewModel.selectAmount(50)
        viewModel.processReload()

        viewModel.uiState.test {
            val state = awaitItem()
            assertThat(state.error).isNotNull()
            assertThat(state.error?.message).contains("Network error")
        }
    }

    @Test
    fun `processReload with zero amount is rejected`() = runTest {
        viewModel.selectAmount(0)
        viewModel.processReload()

        viewModel.uiState.test {
            val state = awaitItem()
            assertThat(state.error).isNotNull()
            coVerify(exactly = 0) { mockRepository.reloadCard(any()) }
        }
    }
}
```

Wery thorough, no? Every path through the code has been charted!"

@HiltAndroidTest
@RunWith(AndroidJUnit4::class)
class FunCardReloadScreenTest {

    @get:Rule(order = 0)
    val hiltRule = HiltAndroidRule(this)

    @get:Rule(order = 1)
    val composeTestRule = createAndroidComposeRule<TestActivity>()

    @Before
    fun setup() {
        hiltRule.inject()
    }

    @Test
    fun reloadScreen_displaysCurrentBalance() {
        composeTestRule.setContent {
            FunCardReloadScreen()
        }

        composeTestRule
            .onNodeWithText("$45.00")
            .assertIsDisplayed()
    }

    @Test
    fun reloadScreen_quickAmountButtons_areClickable() {
        composeTestRule.setContent {
            FunCardReloadScreen()
        }

        listOf("$10", "$25", "$50", "$100").forEach { amount ->
            composeTestRule
                .onNodeWithText(amount)
                .assertIsDisplayed()
                .assertHasClickAction()
        }
    }

    @Test
    fun reloadScreen_selectAmount_updatesButton() {
        composeTestRule.setContent {
            FunCardReloadScreen()
        }

        composeTestRule
            .onNodeWithText("$50")
            .performClick()

        composeTestRule
            .onNodeWithText("Add $50")
            .assertIsDisplayed()
    }

    @Test
    fun reloadScreen_accessibilityCheck() {
        composeTestRule.setContent {
            FunCardReloadScreen()
        }

        // Verify TalkBack descriptions
        composeTestRule
            .onNodeWithContentDescription("Navigate back")
            .assertExists()

        composeTestRule
            .onNodeWithContentDescription("Fun Card reload screen")
            .assertExists()

            .onAllNodesWithTag("quick_amount_button")

    }
}

**GitHub Actions Test Workflow:**

```yaml
name: Android Tests

on:
  pull_request:
    branches: [develop]

jobs:
  unit-tests:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Set up JDK
        uses: actions/setup-java@v4
        with:
          java-version: '17'
          distribution: 'temurin'

      - name: Run Unit Tests
        run: ./gradlew testDebugUnitTest

      - name: Generate Coverage Report
        run: ./gradlew jacocoTestReport

      - name: Upload Coverage
        uses: actions/upload-artifact@v4
        with:
          name: coverage-report
          path: app/build/reports/jacoco/

  lint:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Run ktlint
        run: ./gradlew ktlintCheck

      - name: Run Android Lint
        run: ./gradlew lint
```
